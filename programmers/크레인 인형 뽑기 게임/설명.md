```javascript
function solution(board, moves) {
  const result = [];
  let cnt = 0;
  for (const move of moves) {
    for (let i = 0; i < board.length; i++) {
      if (board[i][move - 1] !== 0) {
        if (
          result.length > 0 &&
          result[result.length - 1] === board[i][move - 1]
        ) {
          result.pop();
          cnt += 2;
        } else {
          result.push(board[i][move - 1]);
        }
        board[i][move - 1] = 0;
        break;
      }
    }
  }
  return cnt;
}
```
